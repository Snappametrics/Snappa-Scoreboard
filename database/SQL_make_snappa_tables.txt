CREATE TABLE players (
	player_id int CONSTRAINT player_key PRIMARY KEY,
	player_name varchar(50));
	
CREATE TABLE scores (
	score_id int,
	game_id int,
	player_id int, 
	scoring_team varchar(1),
	round_num varchar(3),
	points_scored int,
	shooting boolean,
	paddle boolean
);

CREATE TABLE game_stats_players (
	game_id int,
	player_id int,
	team varchar(1),
	total_points int,
	paddle_points int,
	ones int,
	twos int,
	threes int,
	impossibles int,
	points_per_round real;
	shots double precision
	off_ppr double precision,
	def_ppr double precision,
	toss_efficiency double precision
		   	
);

CREATE TABLE games(
	game_id int CONSTRAINT game_key PRIMARY KEY,
	game_start varchar(25),
	game_end varchar(25), 
	A_score int,
	B_score int
);

CREATE TABLE career_stats (
	player_id int CONSTRAINT snappaneer_key PRIMARY KEY,
	player_name varchar(50),
	career_points int,
	career_offense int,
	career_defense int;

CREATE TABLE game_stats (
	game_id int,
	num_players int,
	night_dice boolean,
	points_a int,
	points_b int,
	total_points int,
	total_shots int,
	ones int,
	twos int,
	threes int,
	impossibles int
);	

CREATE TABLE assists (
  game_id int NOT NULL,
  score_id int NOT NULL,
  assist_id int NOT NULL,
  player_id int REFERENCES players (player_id) NOT NULL,
  paddle boolean,
  foot boolean,
  head boolean,
  CONSTRAINT assist_key PRIMARY KEY (game_id, score_id, assist_id),
	FOREIGN KEY (game_id, score_id) REFERENCES scores(game_id, score_id) ON DELETE CASCADE
)
